function openPatientEdit(pid) { const p=getLocalDB().find(x=>x.id===pid); if(p) loadPatientToForm(p); else showToast("❌ غير موجود","error"); } function loadPatientToForm(p) { document.getElementById('editPatientId').value=p.id; document.getElementById('editName').value=p.name||''; document.getElementById('editPhone').value=p.phone||''; document.getElementById('editDiag').value=p.diag||''; document.getElementById('editNextAppt').value=p.nextAppointment||''; navTo('patient'); } function updatePatient() { const id=document.getElementById('editPatientId').value, name=document.getElementById('editName').value.trim(); if(!id||!name) return showToast("الاسم مطلوب","error"); updateLocalDB(id,{name,phone:document.getElementById('editPhone').value.trim(),diag:document.getElementById('editDiag').value.trim()}); showToast("✅ تم التحديث","success"); navTo('list'); updateHomeNotifications(); } function deletePatient() { const id=document.getElementById('editPatientId').value; if(!id||!confirm("حذف المريض؟")) return; deleteFromLocalDB(id); showToast("🗑️ تم الحذف","success"); navTo('list'); updateHomeNotifications(); } function saveNextAppointment() { const id=document.getElementById('editPatientId').value, appt=document.getElementById('editNextAppt').value; if(!id) return; updateLocalDB(id,{nextAppointment:appt}); showToast("✅ تم حفظ الموعد","success"); updateHomeNotifications(); } function filterPatients(q) { const ql=q.toLowerCase().trim(), all = getLocalDB(), filtered = all.filter(p => (p.name||'').toLowerCase().includes(ql) || (p.phone||'').includes(ql)); renderPatients(filtered); } document.addEventListener('DOMContentLoaded', () => { console.log("🦷 بَسْمَة AI جاهز!"); });
